```typescript
interface WorkflowProgressProps {
  currentStage: 'request' | 'quote' | 'job' | 'invoice';
  requestStatus?: string;
  quoteStatus?: string;
  jobStatus?: string;
  invoiceStatus?: string;
}

function WorkflowProgress({ currentStage, ...statuses }: WorkflowProgressProps) {
  const stages = [
    { key: 'request', label: 'Request', status: statuses.requestStatus },
    { key: 'quote', label: 'Quote', status: statuses.quoteStatus },
    { key: 'job', label: 'Job', status: statuses.jobStatus },
    { key: 'invoice', label: 'Invoice', status: statuses.invoiceStatus },
  ];

  return (
    <div className="flex items-center justify-between p-4 bg-gray-50 rounded-lg">
      {stages.map((stage, i) => (
        <React.Fragment key={stage.key}>
          <div className={`flex flex-col items-center ${
            stage.key === currentStage ? 'text-blue-600' : 'text-gray-400'
          }`}>
            <div className={`w-8 h-8 rounded-full flex items-center justify-center ${
              getStageColor(stage.status)
            }`}>
              {getStageIcon(stage.status)}
            </div>
            <span className="text-xs mt-1">{stage.label}</span>
            <span className="text-xs opacity-60">{stage.status || 'N/A'}</span>
          </div>
          {i < stages.length - 1 && (
            <div className={`flex-1 h-0.5 mx-2 ${
              hasProgressed(stage.status) ? 'bg-green-400' : 'bg-gray-300'
            }`} />
          )}
        </React.Fragment>
      ))}
    </div>
  );
}
```

## Button Hierarchy

### Primary Action Button
- One per page
- Large, prominent
- Colored (blue or green)
- Represents the NEXT step in workflow

```typescript
<button className="w-full py-3 px-6 bg-blue-600 text-white text-lg font-semibold rounded-lg hover:bg-blue-700 shadow-md">
  <ArrowRight className="w-5 h-5 inline mr-2" />
  Create Quote
</button>
```

### Secondary Action Buttons
- Smaller
- Outlined or subtle
- For common but non-primary actions

```typescript
<div className="flex gap-2">
  <button className="px-4 py-2 text-gray-700 border rounded-lg hover:bg-gray-50">
    <Edit className="w-4 h-4 inline mr-1" />
    Edit
  </button>
  <button className="px-4 py-2 text-gray-700 border rounded-lg hover:bg-gray-50">
    <FileText className="w-4 h-4 inline mr-1" />
    View PDF
  </button>
</div>
```

### Destructive Actions
- Red color
- Requires confirmation
- Usually in dropdown menu

```typescript
<button 
  className="px-4 py-2 text-red-600 hover:bg-red-50 rounded-lg"
  onClick={() => setShowDeleteConfirm(true)}
>
  <Trash className="w-4 h-4 inline mr-1" />
  Delete
</button>
```
